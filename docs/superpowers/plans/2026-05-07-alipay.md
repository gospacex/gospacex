> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 在 gpx 脚手架生成器中添加 `--play alipay` 参数，生成带支付宝支付功能的微服务项目。

**架构：** 支付宝功能注入到 `--srvs` 指定的服务模块中，SRV 层实现业务逻辑（生成支付链接、处理回调），BFF 层通过 rpcClient 调用 SRV 接口，HTTP 接口由 handler 直接暴露。

**技术栈：** Go, Cobra CLI, gopay/gopay, Protobuf/gRPC, Gin

```go
package alipay

import (
    "context"
    "fmt"

    "github.com/go-pay/gopay"
    "github.com/go-pay/gopay/alipay"
)

type Service struct {
    client *alipay.Client
}

func NewService() (*Service, error) {
    client, err := alipay.NewClient(AppID, PrivateKey, IsProduction)
    if err != nil {
        return nil, fmt.Errorf("failed to create alipay client: %w", err)
    }
    client.SetAliPayPublicKey(AliPublicKey)
    return &Service{client: client}, nil
}

// GeneratePayURL 生成支付宝支付链接
func (s *Service) GeneratePayURL(ctx context.Context, orderId, totalAmount, subject string) (string, error) {
    bm := gopay.BodyMap{}
    bm.Set("out_trade_no", orderId)
    bm.Set("total_amount", totalAmount)
    bm.Set("subject", subject)
    bm.Set("product_code", ProductCode)
    bm.Set("notify_url", NotifyURL)

    payUrl, err := s.client.TradePagePay(ctx, bm)
    if err != nil {
        return "", fmt.Errorf("failed to generate pay url: %w", err)
    }
    return payUrl, nil
}

// HandleNotify 处理支付宝回调
func (s *Service) HandleNotify(ctx context.Context, params map[string]string) (string, error) {
    ok, err := s.client.VerifySign(params)
    if err != nil || !ok {
        return "failure", fmt.Errorf("signature verification failed: %w", err)
    }
    // TODO: 更新订单状态
    return "success", nil
}
```

```go
package handler

import (
    "net/http"

    "github.com/gin-gonic/gin"
    "{{.AppName}}/internal/rpcClient"
)

type OrderHandler struct {
    orderRpcClient *rpcClient.OrderRpcClient
}

func NewOrderHandler(client *rpcClient.OrderRpcClient) *OrderHandler {
    return &OrderHandler{orderRpcClient: client}
}

// GeneratePayURLHandler 生成支付链接
func (h *OrderHandler) GeneratePayURL(c *gin.Context) {
    var req struct {
        OrderId     string `json:"orderId"`
        TotalAmount string `json:"totalAmount"`
        Subject     string `json:"subject"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    payUrl, err := h.orderRpcClient.GeneratePayURL(c.Request.Context(), req.OrderId, req.TotalAmount, req.Subject)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, gin.H{"payUrl": payUrl})
}

// HandleNotifyHandler 处理支付宝回调
func (h *OrderHandler) HandleNotify(c *gin.Context) {
    if err := c.Request.ParseForm(); err != nil {
        c.String(http.StatusBadRequest, "failure")
        return
    }

    params := make(map[string]string)
    for key, values := range c.Request.Form {
        if len(values) > 0 {
            params[key] = values[0]
        }
    }

    status, err := h.orderRpcClient.HandleNotify(c.Request.Context(), params)
    if err != nil {
        c.String(http.StatusInternalServerError, "failure")
        return
    }

    c.String(http.StatusOK, status)
}
```

```go
package rpcClient

import (
    "context"
    "fmt"

    "google.golang.org/grpc"
    pb "{{.AppName}}/common/alipay"
)

type OrderRpcClient struct {
    client pb.AlipayServiceClient
}

func NewOrderRpcClient(conn *grpc.ClientConn) *OrderRpcClient {
    return &OrderRpcClient{
        client: pb.NewAlipayServiceClient(conn),
    }
}

func (c *OrderRpcClient) GeneratePayURL(ctx context.Context, orderId, totalAmount, subject string) (string, error) {
    req := &pb.GeneratePayURLRequest{
        OrderId:     orderId,
        TotalAmount: totalAmount,
        Subject:     subject,
    }
    resp, err := c.client.GeneratePayURL(ctx, req)
    if err != nil {
        return "", fmt.Errorf("rpc GeneratePayURL failed: %w", err)
    }
    return resp.PayUrl, nil
}

func (c *OrderRpcClient) HandleNotify(ctx context.Context, params map[string]string) (string, error) {
    req := &pb.HandleNotifyRequest{
        Params: params,
    }
    resp, err := c.client.HandleNotify(ctx, req)
    if err != nil {
        return "failure", fmt.Errorf("rpc HandleNotify failed: %w", err)
    }
    return resp.Status, nil
}
```

```go
func (g *MicroAppGenerator) injectAlipayCode() error {
    // 为每个 srv 模块注入支付宝代码
    for _, m := range g.config.Modules {
        // 生成 alipay 配置和服务文件

            return err

            return err
        }
        // 生成 proto 文件
        if err := g.renderTemplate("microservice/api/alipay/alipay.proto.tmpl", ...); err != nil {

        }
    }

    // 为 BFF 注入 handler 和 rpcClient

        return err
    }
    if err := g.renderTemplate("micro-bff/internal/rpcClient/orderRpcClient.go.tmpl", ...); err != nil {
```
